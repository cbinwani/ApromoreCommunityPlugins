21  package org.apromore.plugin.portal.bpmnminer;
22  
23  import com.raffaeleconforti.foreignkeydiscovery.functionaldependencies.NoEntityException;
24  import org.zkoss.zk.ui.event.Event;
25  import org.zkoss.zk.ui.event.EventListener;
26  import org.zkoss.zul.Button;
27  import org.zkoss.zul.Listbox;
28  import org.zkoss.zul.Listitem;
29  import org.zkoss.zul.Window;
30  
31  import java.io.IOException;
32  import java.util.List;
33  
34  /**
35   * Created by conforti on 9/04/15.
36   */
37  public class CandidatesEntitiesController {
38  
39      private static final long serialVersionUID = 1L;
40      private BPMNMinerController bpmnMinerController;
41      private Window candidateEntitiesW;
42      private Listbox candidates;
43      private Button cancelButton;
44      private Button nextButton;
45  
46      private List<String> listCandidates;
47      private boolean[] selected;
48  
49      public CandidatesEntitiesController(BPMNMinerController bpmnMinerController, List<String> listCandidates) throws IOException {
50          this.bpmnMinerController = bpmnMinerController;
51          this.selected = new boolean[listCandidates.size()];
52          this.listCandidates = listCandidates;
53  
54          this.candidateEntitiesW = (Window) bpmnMinerController.portalContext.getUI().createComponent(getClass().getClassLoader(), "zul/candidateEntities.zul", null, null);
55          this.candidateEntitiesW.setTitle("Select Candidate Entities");
56  
57          this.candidates = (Listbox) this.candidateEntitiesW.getFellow("list");
58          for(String candidate : listCandidates) {
59              Listitem listItem = new Listitem();
60              listItem.setLabel(candidate);
61              this.candidates.appendChild(listItem);
62              listItem.setSelected(false);
63          }
64  
65          this.cancelButton = (Button) this.candidateEntitiesW.getFellow("entityCancelButton");
66          this.nextButton = (Button) this.candidateEntitiesW.getFellow("entityNextButton");
67  
68          updateActions();
69  
70          this.candidates.addEventListener("onSelect", new EventListener<Event>() {
71              public void onEvent(Event event) throws Exception {
72                  updateActions();
73              }
74          });
75          this.cancelButton.addEventListener("onClick", new EventListener<Event>() {
76              public void onEvent(Event event) throws Exception {
77                  cancel();
78              }
79          });
80          this.nextButton.addEventListener("onClick", new EventListener<Event>() {
81              public void onEvent(Event event) throws Exception {
82                  next();
83              }
84          });
85  
86          if(listCandidates != null && listCandidates.size() > 0) {
87              this.candidateEntitiesW.doModal();
88          }else {
89              next();
90          }
91      }
92  
93      protected void cancel() {
94          this.candidateEntitiesW.detach();
95      }
96  
97      protected void next() {
98          this.candidateEntitiesW.detach();
99          try {
100             this.bpmnMinerController.setSelectedCandidatesEntities(listCandidates, selected);
101         } catch (IOException | NoEntityException e) {
102             this.bpmnMinerController.noEntityException();
103         }
104     }
105 
106     protected void updateActions() {
107         for(int i = 0; i < selected.length; i++) {
108             boolean done = false;
109             for(Listitem listItem : candidates.getSelectedItems()) {
110                 if(listCandidates.get(i).equals(listItem.getLabel())) {
111                     selected[i] = true;
112                     done = true;
113                     break;
114                 }
115             }
116             if(!done) selected[i] = false;
117         }
118     }
119 }
