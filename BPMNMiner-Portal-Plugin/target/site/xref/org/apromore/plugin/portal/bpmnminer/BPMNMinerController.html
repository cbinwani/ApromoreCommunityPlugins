248     public static XLog importFromStream(XFactory factory, InputStream is, String name) throws Exception {
249         XParser parser = null;
250         if(name.endsWith("mxml")) {
251             parser = new XMxmlParser(factory);
252         }else if(name.endsWith("mxml.gz")) {
253             parser = new XMxmlGZIPParser(factory);
254         }else if(name.endsWith("xes")) {
255             parser = new XesXmlParser(factory);
256         }else if(name.endsWith("xes.gz")) {
257             parser = new XesXmlGZIPParser(factory);
258         }
259 
260         Collection<XLog> logs;
261         try {
262             logs = parser.parse(is);
263         } catch (Exception e) {
264             e.printStackTrace();
265             logs = null;
266         }
267         if (logs == null) {
268             // try any other parser
269             for (XParser p : XParserRegistry.instance().getAvailable()) {
270                 if (p == parser) {
271                     continue;
272                 }
273                 try {
274                     logs = p.parse(is);
275                     if (logs.size() > 0) {
276                         break;
277                     }
278                 } catch (Exception e1) {
279                     // ignore and move on.
280                     logs = null;
281                 }
282             }
283         }
284 
285         // log sanity checks;
286         // notify user if the log is awkward / does miss crucial information
287         if (logs == null || logs.size() == 0) {
288             throw new Exception("No processes contained in log!");
289         }
290 
291         XLog log = logs.iterator().next();
292         if (XConceptExtension.instance().extractName(log) == null) {
293             XConceptExtension.instance().assignName(log, "Anonymous log imported from ");
294         }
295 
296         if (log.isEmpty()) {
297             throw new Exception("No process instances contained in log!");
298         }
299 
300         return log;
301 
302     }

317     protected void createCanditatesEntity() {
318         try {
319             if(log == null) {
320                 ByteArrayOutputStream bos = new ByteArrayOutputStream();
321                 bos.write(logByteArray, 0, logByteArray.length);
322                 InputStream zipEntryIS = new ByteArrayInputStream(bos.toByteArray());
323                 log = importFromStream(new XFactoryNaiveImpl(), zipEntryIS, logFileName);
324             }
325 
326             if(log == null) {
327                 Messagebox.show("Please select a log.");
328             }else {
329                 if(flatModel.getSelectedIndex() == 0) {
330                     noEntityException();
331                 }else {
332                     erModel = new DiscoverERmodel();
333                     listCandidates = erModel.generateAllAttributes(log);
334                     Collections.sort(listCandidates);
335 
336                     new CandidatesEntitiesController(this, listCandidates);
337                 }
338             }
339 
340         } catch (IOException e) {
341             e.printStackTrace();
342         } catch (Exception e) {
343             e.printStackTrace();
344         }
345     }

373     private void mineAndSave(List<String> listCandidates, Map<Set<String>, Set<String>> group) {
374         try {
375 
376             this.bpmnMinerW.detach();
377 
378             if(filterLog.getSelectedIndex() == 0) {
379                 log = infrequentBehaviourFilterService.filterLog(log);
380             }
381 
382             String model = bpmnMinerService.discoverBPMNModel(log, sortLog.getSelectedIndex()==0?true:false, structProcess.getSelectedIndex()==0?true:false, getSelectedAlgorithm(), params, dependencyAlgorithms.getSelectedIndex()+1,
383                     ((double) interruptingEventTolerance.getCurpos())/100.0, ((double) timerEventPercentage.getCurpos())/100.0, ((double) timerEventTolerance.getCurpos())/100.0,
384                     ((double) multiInstancePercentage.getCurpos())/100.0, ((double) multiInstanceTolerance.getCurpos())/100.0, ((double) noiseThreshold.getCurpos())/100.0,
385                     listCandidates, group);
386 
387             String defaultProcessName = null;
388             if(this.logFileName != null) {
389                 defaultProcessName = this.logFileName.split("\\.")[0];
390             }
391             if (!modelName.getValue().isEmpty()) {
392                 defaultProcessName = modelName.getValue();
393             }
394 
395             String user = portalContext.getCurrentUser().getUsername();
396             Version version = new Version(1, 0);
397             Set<RequestParameterType<?>> canoniserProperties = new HashSet<>();
398             String now = DatatypeFactory.newInstance().newXMLGregorianCalendar(new GregorianCalendar()).toString();
399             boolean publicModel = true;
400             
401             ProcessModelVersion pmv = processService.importProcess(user,
402                 portalContext.getCurrentFolder() == null ? 0 : portalContext.getCurrentFolder().getId(),
403                 defaultProcessName,
404                 version,
405                 this.nativeType,
406                 canoniserService.canonise(this.nativeType, new ByteArrayInputStream(model.getBytes()), canoniserProperties),
407                 domainCB.getValue(),
408                 "Model generated by the Apromore BPMN process mining service.",
409                 now,  // creation timestamp
410                 now,  // last update timestamp
411                 publicModel);
412 
413             this.portalContext.displayNewProcess(userInterfaceHelper.createProcessSummary(pmv.getProcessBranch().getProcess(),
414                 pmv.getProcessBranch(),
415                 pmv,
416                 this.nativeType,
417                 domainCB.getValue(),
418                 now,  // creation timestamp
419                 now,  // last update timestamp
420                 user,
421                 publicModel));
422 
423             this.portalContext.refreshContent();
424 
425         } catch (Exception e) {
426             e.printStackTrace();
427         }
428     }

430     public int getSelectedAlgorithm() {
431         int selected = miningAlgorithms.getSelectedIndex();
432         String name = "";
433 
434         switch (selected) {
435             case 0 : name = SelectMinerResult.SM; break;
436             case 1 : name = SelectMinerResult.IM; break;
437             case 2 : name = SelectMinerResult.HM6; break;
438             case 3 : name = SelectMinerResult.HM5; break;
439             case 4 : name = SelectMinerResult.ALPHA; break;
440             case 5 : name = SelectMinerResult.ILP;
441         }
442 
443         if(name.equals(SelectMinerResult.ALPHA)) return SelectMinerResult.ALPHAPOS;
444         else if(name.equals(SelectMinerResult.HM6)) return SelectMinerResult.HMPOS6;
445         else if(name.equals(SelectMinerResult.HM5)) return SelectMinerResult.HMPOS5;
446         else if(name.equals(SelectMinerResult.SM)) return SelectMinerResult.SMPOS;
447         else if(name.equals(SelectMinerResult.ILP)) return SelectMinerResult.ILPPOS;
448         else if(name.equals(SelectMinerResult.IM)) return SelectMinerResult.IMPOS;
449 
450         return SelectMinerResult.SMPOS;
451     }
